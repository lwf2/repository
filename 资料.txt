FROM
	T_ORDER A
WHERE
	(A.OUT_ORDER_ID) IN (
		SELECT
			OUT_ORDER_ID
		FROM
			T_ORDER
		GROUP BY
			OUT_ORDER_ID
		HAVING
			COUNT (*) > 1
	)
AND ROWID NOT IN (
	SELECT
		MIN (ROWID)
	FROM
		T_ORDER
	GROUP BY
		OUT_ORDER_ID
	HAVING
		COUNT (*) > 1
)
